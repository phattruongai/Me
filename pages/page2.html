    @staticmethod
    def text_to_coor(page,h,w):
        word_dict = {} 
        for idx,word in enumerate(page.lines):
            x1 = int(word.geometry.boundingBox.left * w)
            y1 = int(word.geometry.boundingBox.top * h)
            x2 = int(word.geometry.boundingBox.right * w)
            y2 = int(word.geometry.boundingBox.bottom * h)
            word_dict[f"e{idx}"] = {"coor":[x1,y1,x2,y2],"text":word.text.replace("*","").strip()}
        text_coor_meta = {}
        text_coor_meta["rawTextData"] = word_dict
        return text_coor_meta
def inside(word_box, user_box):
    """
    Match user draw boxes with output boxes from textract
    """
    # determine the (x, y)-coordinates of the intersection rectangle
    xA = max(word_box[0], user_box[0])
    yA = max(word_box[1], user_box[1])
    xB = min(word_box[2], user_box[2])
    yB = min(word_box[3], user_box[3])
    # compute the area of intersection rectangle
    interArea = max(0, xB - xA + 1) * max(0, yB - yA + 1)
    # compute the area of both the prediction and ground-truth
    # rectangles
    word_boxArea = (word_box[2] - word_box[0] + 1) * (word_box[3] - word_box[1] + 1)
    user_boxArea = (user_box[2] - user_box[0] + 1) * (user_box[3] - user_box[1] + 1)
    # compute the intersection over union by taking the intersection
    # area and dividing it by the sum of prediction + ground-truth
    # areas - the interesection area
    inside_rate = interArea / float(word_boxArea)
    # return the intersection over union value
    return inside_rate
